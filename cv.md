function dieWrapper(operation, a, b) {
    if (Array.isArray(a) && Array.isArray(b)) {
        const res = Array(a.length * b.length);
        for (let i = 0; i < a.length; i++) {
            for (let j = 0; j < b.length; j++) {

        }
